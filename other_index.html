import matplotlib.pyplot as plt

 

# x axis values

x = [1,2,3]

# corresponding y axis values

y = [2,4,1]

 

plt.figure() # creates a new figure to graph on

plt.plot(x, y) # graphs x and y against each other on the new figure

plt.show() #displays the figure
